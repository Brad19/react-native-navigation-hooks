## NavigationContext

### `NavigationProvider`

```tsx
import React, { ReactElement } from 'react'
import { Navigation } from 'react-native-navigation'
import { NavigationProvider } from 'react-native-navigation-hooks'
import { MyScreenComponent } from '~/screens'

Navigation.registerComponent(
  'MyScreenComponent',
  () => (props): ReactElement => {
    return (
      <NavigationProvider>
        <MyScreenComponent {...props} />
      </NavigationProvider>>
    )
  },
  () => MyScreenComponent
)
```

### `NavigationConsumer`

```tsx
import React, { ReactElement } from 'react'
import { NavigationConsumer } from 'react-native-navigation-hooks'

const MyComponent = () => {
  return <NavigationConsumer>{({ componentId }) => <Text>{componentId}</Text>}</NavigationConsumer>
}
```

or you can use React's `useContext` hook to access navigation context values.

```tsx
import { useContext } from 'react'
import { NavigationContext } from 'react-native-navigation-hooks'

const MyComponent = () => {
  const { componentId } = useContext(NavigationContext)
}
```
